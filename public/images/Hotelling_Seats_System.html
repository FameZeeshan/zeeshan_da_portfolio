DATE_FORMAT = '%d-%b-%Y'

class HotellingSeatsSystem:
    """
    Complete Hotelling Seats Management System
    Manages resource allocation, conflict detection, and occupancy analysis
    """
    
    def __init__(self):
        self.port_config = []
        self.allocations = []
    
    def load_data(self, filepath, ports_sheet='Ports', allocations_sheet='Allocations'):
        """
        Load ports and allocations from Excel file.
        
        Args:
            filepath: Excel file path
            ports_sheet: Name of ports sheet
            allocations_sheet: Name of allocations sheet
        """
        try:
            # Load ports
            df_ports = pd.read_excel(filepath, sheet_name=ports_sheet)
            port_dict = {}
            for _, row in df_ports.iterrows():
                city = str(row['City']).strip()
                port = str(row['Port']) if not pd.isna(row['Port']) else ''
                port_dict.setdefault(city, []).append(port)
            
            for city, ports in port_dict.items():
                for port in ports:
                    if not any(p['Port'] == port and p['City'] == city for p in self.port_config):
                        self.port_config.append({'City': city, 'Port': port})
            
            # Load allocations
            df_alloc = pd.read_excel(filepath, sheet_name=allocations_sheet)
            for _, row in df_alloc.iterrows():
                city = str(row['City']).strip()
                ports = [p.strip().strip("'").strip('"') for p in str(row['Ports']).split(',')]
                program = str(row['Program']).strip()
                start_date = str(row['Start']).strip()
                end_date = str(row['End']).strip()
                
                # Validate and add allocation
                self._validate_and_add_allocation(city, ports, program, start_date, end_date)
        
        except Exception as e:
            print(f"Error loading data: {str(e)}")
            raise
    
    def _validate_and_add_allocation(self, city, ports, program, start_date, end_date):
        """
        Validate and add allocation with conflict detection.
        """
        start = datetime.strptime(start_date, DATE_FORMAT)
        end = datetime.strptime(end_date, DATE_FORMAT)
        
        # Check duplicates in ports list
        duplicates = [port for port in set(ports) if ports.count(port) > 1]
        if duplicates:
            raise ValueError(
                f"Duplicate ports in {program}: {', '.join(duplicates)}"
            )
        
        # Validate ports exist
        for port in ports:
            if not any(p['Port'] == port and p['City'] == city for p in self.port_config):
                raise ValueError(f"Port '{port}' not configured for city '{city}'")
        
        # Check conflicts
        for port in ports:
            for alloc in self.allocations:
                if alloc['City'] == city and alloc['Port'] == port:
                    existing_start = datetime.strptime(alloc['Start'], DATE_FORMAT)
                    existing_end = datetime.strptime(alloc['End'], DATE_FORMAT)
                    if (start <= existing_end) and (end >= existing_start):
                        raise ValueError(
                            f"Conflict: Port '{port}' allocated to '{alloc['Program']}' "
                            f"({alloc['Start']} to {alloc['End']})"
                        )
        
        # Add allocation
        for port in ports:
            self.allocations.append({
                'City': city,
                'Port': port,
                'Program': program,
                'Start': start_date,
                'End': end_date
            })
    
    def get_allocations(self):
        """Return all allocations as DataFrame."""
        return pd.DataFrame(self.allocations)
    
    def get_port_config(self):
        """Return port configuration as DataFrame."""
        return pd.DataFrame(self.port_config)
    
    def check_port_availability(self, city, port, start_date, end_date):
        """
        Check if a port is available for date range.
        
        Returns:
            bool: True if available, False otherwise
        """
        start = datetime.strptime(start_date, DATE_FORMAT)
        end = datetime.strptime(end_date, DATE_FORMAT)
        
        for alloc in self.allocations:
            if alloc['City'] == city and alloc['Port'] == port:
                existing_start = datetime.strptime(alloc['Start'], DATE_FORMAT)
                existing_end = datetime.strptime(alloc['End'], DATE_FORMAT)
                if (start <= existing_end) and (end >= existing_start):
                    return False
        return True
    
    def get_occupancy_info(self, city, from_date, to_date):
        """
        Get occupancy information for a city and date range.
        
        Returns:
            tuple: (program_occupancy_dict, free_ports_list)
        """
        from_dt = datetime.strptime(from_date, DATE_FORMAT)
        to_dt = datetime.strptime(to_date, DATE_FORMAT)
        
        city_allocs = [a for a in self.allocations if a['City'] == city]
        program_occupancy = {}
        allocated_ports = set()
        
        for alloc in city_allocs:
            alloc_start = datetime.strptime(alloc['Start'], DATE_FORMAT)
            alloc_end = datetime.strptime(alloc['End'], DATE_FORMAT)
            
            if (alloc_start <= to_dt) and (alloc_end >= from_dt):
                program_occupancy[alloc['Program']] = program_occupancy.get(alloc['Program'], 0) + 1
                allocated_ports.add(alloc['Port'])
        
        all_ports = {p['Port'] for p in self.port_config if p['City'] == city}
        free_ports = sorted(list(all_ports - allocated_ports))
        
        return program_occupancy, free_ports
    
    def get_summary_report(self, city, from_date, to_date):
        """
        Generate summary report for city and date range.
        
        Returns:
            pd.DataFrame: Summary with programs and free seats
        """
        program_occupancy, free_ports = self.get_occupancy_info(city, from_date, to_date)
        total_seats = len(free_ports) + sum(program_occupancy.values())
        
        summary_data = []
        for program, count in program_occupancy.items():
            summary_data.append({
                'Name': program,
                'Occupied Seats': count,
                'Percentage': f"{(count / total_seats * 100):.1f}%" if total_seats > 0 else "0%"
            })
        
        summary_data.append({
            'Name': 'Free Seats',
            'Occupied Seats': len(free_ports),
            'Percentage': f"{(len(free_ports) / total_seats * 100):.1f}%" if total_seats > 0 else "0%"
        })
        
        return pd.DataFrame(summary_data)
    
    def export_allocations(self, filename='Hotelling_Seats_Report.xlsx'):
        """
        Export allocations to Excel file.
        """
        df = self.get_allocations()
        df.to_excel(filename, index=False)
    
    def visualize_occupancy(self, city, year, month, logo_path=None):
        """
        Create occupancy heatmap visualization.
        
        Args:
            city: Target city
            year: Year for analysis
            month: Month for analysis
            logo_path: Optional path to company logo
        """
        allocations_df = self.get_allocations().copy()
        ports_df = self.get_port_config().copy()
        
        # Filter by city
        allocations_df = allocations_df[allocations_df['City'] == city]
        ports_df = ports_df[ports_df['City'] == city]
        
        if allocations_df.empty:
            print(f"No allocations found for {city}")
            return
        
        # Convert dates
        allocations_df['Start'] = pd.to_datetime(allocations_df['Start'], format=DATE_FORMAT)
        allocations_df['End'] = pd.to_datetime(allocations_df['End'], format=DATE_FORMAT)
        
        # Create date range for month
        start_date = datetime(year, month, 1)
        end_date = (start_date + timedelta(days=32)).replace(day=1) - timedelta(days=1)
        date_range = pd.date_range(start=start_date, end=end_date, freq='D')
        
        # Build occupancy matrix
        programs = allocations_df['Program'].unique()
        occupancy_df = pd.DataFrame(0, index=date_range, columns=programs, dtype=int)
        
        for _, row in allocations_df.iterrows():
            overlap_start = max(row['Start'], start_date)
            overlap_end = min(row['End'], end_date)
            if overlap_start <= overlap_end:
                occupancy_df.loc[overlap_start:overlap_end, row['Program']] += 1
        
        # Filter to programs with allocations
        occupancy_df = occupancy_df.loc[:, (occupancy_df != 0).any(axis=0)]
        
        total_ports = ports_df['Port'].nunique()
        occupancy_df['Free Seats'] = total_ports - occupancy_df.sum(axis=1)
        occupancy_df.index = occupancy_df.index.strftime('%d-%b')
        
        # Prepare data for heatmap
        program_df = occupancy_df.drop(columns=['Free Seats']).T
        free_seats_df = occupancy_df[['Free Seats']].T
        combined_df = pd.concat([program_df, free_seats_df], axis=0)
        
        # Create visualization
        fig, ax = plt.subplots(figsize=(16, 8))
        sns.heatmap(combined_df, ax=ax,
                    cmap=sns.light_palette('blue', as_cmap=True),
                    cbar_kws={'label': 'Count'},
                    linewidths=0.5,
                    annot=True,
                    fmt='.0f',
                    vmin=0,
                    vmax=total_ports)
        
        # Title and labels
        month_name = start_date.strftime('%b %Y')
        title = f"Hotelling Seats Occupancy Heatmap - {city}, {month_name}"
        fig.suptitle(title, fontsize=14, y=0.98, ha='center', va='top')
        
        # Add logo if provided
        if logo_path:
            try:
                img = Image.open(logo_path)
                oi = OffsetImage(img, zoom=0.22)
                ab = AnnotationBbox(oi, (0.01, 0.995),
                                   xycoords='figure fraction',
                                   xybox=(2, 2),
                                   boxcoords='offset points',
                                   box_alignment=(0, 1),
                                   frameon=False,
                                   zorder=100)
                fig.add_artist(ab)
            except:
                pass
        
        ax.set_xlabel('Date', fontsize=11)
        ax.set_ylabel('Programs / Free Seats', fontsize=11)
        ax.set_xticklabels(ax.get_xticklabels(), rotation=45, ha='right')
        
        # Separator line
        n_program_rows = program_df.shape[0]
        ax.hlines(n_program_rows, *ax.get_xlim(), colors='black', linewidth=1.5)
        
        # Color free seats label
        ylabels = ax.get_yticklabels()
        if ylabels:
            ylabels[-1].set_color('green')
        
        plt.tight_layout()
        filename = f"Occupancy_{city}_{start_date.strftime('%b_%Y')}.png"
        plt.savefig(filename, dpi=150, bbox_inches='tight')
        plt.show()
